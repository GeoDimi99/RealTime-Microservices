
# New commands 
# Build
sudo docker build -t execution-manager -f services/execution-manager/Dockerfile .

cat services/task-wrapper/src/app_task_sum.c > services/task-wrapper/src/app_task.c
sudo docker build --build-arg TASK_NAME=sum -t sum -f services/task-wrapper/Dockerfile .

cat services/task-wrapper/src/app_task_subtract.c > services/task-wrapper/src/app_task.c
sudo docker build --build-arg TASK_NAME=subtract -t subtract -f services/task-wrapper/Dockerfile .

cat services/task-wrapper/src/app_task_multiply.c > services/task-wrapper/src/app_task.c
sudo docker build --build-arg TASK_NAME=multiply -t multiply -f services/task-wrapper/Dockerfile .


# Run
sudo docker run --rm --ipc=host --name execution-manager execution-manager:latest 

sudo docker run --rm --ipc=host --cap-add=SYS_NICE --ulimit rtprio=99 -e TASK_NAME=sum -e TASK_QUEUE_NAME=sum --cap-add=IPC_LOCK --ulimit memlock=-1:-1 --name sum sum:latest

sudo docker run --rm --ipc=host --cap-add=SYS_NICE --ulimit rtprio=99 -e TASK_NAME=subtract -e TASK_QUEUE_NAME=subtract --cap-add=IPC_LOCK --ulimit memlock=-1:-1 --name subtract subtract:latest

sudo docker run --rm --ipc=host --cap-add=SYS_NICE --ulimit rtprio=99 -e TASK_NAME=multiply -e TASK_QUEUE_NAME=multiply --cap-add=IPC_LOCK --ulimit memlock=-1:-1 --name multiply multiply:latest
